for x in 12 ... 18 do

    bmi := x * 2.3;

    if bmi is greater than or equal 40 then
        trace "Adipositas III";
    elseif bmi is greater than or equal 35 then
        trace "Adipositas II";
    elseif bmi is greater than or equal 30 then
        trace "Adipositas I";
    else
        trace "Keine Adipositas";
    endif;
enddo;
